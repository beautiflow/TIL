    public Cat(){

    }
    
    // 개발자가 직접 생성자를 구현함 = 디폴트 생성자가 생기지 않음.
    public Cat(String n, String c){  // 생성자 (메서드)

      System.out.println("n :" + n );

      System.out.println("c :" + c );
      
       //String n, String c 는 지역변수 = Cat() 이라는 생성자(메서드)가 가진 지역변수
      name = n;
      color = c;

    }
}

public class CatApp{
